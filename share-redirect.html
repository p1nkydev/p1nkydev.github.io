    <script>

        function getMobileOS() {

            const userAgent = navigator.userAgent || navigator.vendor || window.opera;

            

            // iOS detection

            if (/iPad|iPhone|iPod/.test(userAgent) && !window.MSStream) {

                return "iOS";

            }

            

            // Android detection

            if (/android/i.test(userAgent)) {

                return "Android";

            }

            

            return "unknown";

        }

        

        function redirect() {

            const urlParams = new URLSearchParams(window.location.search);

            const contentId = urlParams.get('id') || '';

            const os = getMobileOS();

            

            let appDeepLink = '';

            let storeLink = '';

            

            if (os === "Android") {

                appDeepLink = `https://p1nkydev.github.io/share?id=${contentId}`;

                storeLink = 'https://play.google.com/store/apps/details?id=ua.companyname.porada.android';

                

            } else if (os === "iOS") {

                // iOS deep link

                appDeepLink = `https://p1nkydev.github.io/share?id=${contentId}`;

                storeLink = 'https://apps.apple.com/app/com.companyname.porada';

                

            } else {

                // Desktop or unknown - show message or redirect to website

                document.body.innerHTML = '<div class="container"><p>Please open this link on your mobile device</p></div>';

                return;

            }

            

            // Attempt to open the app

            window.location.href = appDeepLink;

            

            // Fallback to store if app didn't open

            let appOpened = false;

            

            const handleVisibilityChange = () => {

                if (document.hidden) {

                    appOpened = true;

                }

            };

            

            document.addEventListener('visibilitychange', handleVisibilityChange);

            

            // Redirect to store after delay if app didn't open

            setTimeout(() => {

                document.removeEventListener('visibilitychange', handleVisibilityChange);

                if (!appOpened) {

                    window.location.href = storeLink;

                }

            }, 2000);

        }

        

        window.onload = redirect;

    </script>
